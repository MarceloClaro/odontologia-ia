% ---- QR Code do Capítulo ----
\newcommand{\qrchapter}[4]{%
  \begin{center}
  \fcolorbox{CorQRCode}{white}{%
    \begin{minipage}{0.9\textwidth}
    \small
    \textbf{📱 Conteúdo Digital do Capítulo}
    \hfill
    \includegraphics[width=1.5cm]{ilustracoes/qrcode-modelo.png}
    
    \textbf{#1 — #2}
    
    \begin{itemize}
      \item \textbf{Vídeo:} \texttt{#3} 
      \item \textbf{Colab:} \texttt{#4}
    \end{itemize}
    
    \textit{Escaneie o QR code ao lado ou acesse: 
    \url{https://odontologia-ia.edsonlaranjeiras.com.br/cap#1}}
    \end{minipage}%
  }
  \end{center}
}
